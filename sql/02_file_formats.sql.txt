--FILE FORMATS
--CSV FORMAT
CREATE OR REPLACE FILE FORMAT BRONZE.CSV_FMT
TYPE = 'CSV'
FIELD_DELIMITER = ','
SKIP_HEADER = 1
EMPTY_FIELD_AS_NULL = TRUE
date_format = '%D-%M-%Y'
ERROR_ON_COLUMN_COUNT_MISMATCH = FALSE
NULL_IF = ('NULL','null','')
TRIM_SPACE = TRUE;

--JSON FORMAT
CREATE OR REPLACE FILE FORMAT BRONZE.JSON_FMT
TYPE = JSON;
