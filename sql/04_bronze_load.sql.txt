--CREATING RAW TABLES

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.CUSTOMER_DETAILS_RAW (
  customer_id STRING,
  first_name STRING,
  last_name STRING,
  email STRING,
  phone STRING,
  city STRING,
  state STRING,
  loyalty_tier STRING,
  created_at TIMESTAMP_NTZ
);

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.PRODUCTS_RAW (
  product_id STRING,
  product_name STRING,
  category STRING,
  price NUMBER(10,2),
  brand STRING
);

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.STORES_RAW (
  store_id STRING,
  store_name STRING,
  city STRING,
  state STRING,
  region STRING
);

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.STORE_SALES_HEADER_RAW (
  transaction_id STRING,
  customer_id STRING,
  store_id STRING,
  transaction_timestamp TIMESTAMP_NTZ,
  total_amount NUMBER(12,2)
);

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.STORE_SALES_LINE_ITEMS_RAW (
  line_item_id NUMBER,
  transaction_id STRING,
  product_id STRING,
  quantity NUMBER,
  price NUMBER(10,2)
);

CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.LOYALTY_RULES_RAW (raw VARIANT);
CREATE OR REPLACE TABLE RETAILPULSE_DB.BRONZE.PROMOTION_DETAILS_RAW (raw VARIANT);

--CUSTOMERS

COPY INTO RETAILPULSE_DB.BRONZE.CUSTOMER_DETAILS_RAW
FROM
(
SELECT
$1,$2,$3,$4,$5,$6,$7,$8,$9,
METADATA$FILENAME
FROM @RETAILPULSE_DB.BRONZE.S3_CSV_STAGE/customers/ 
)
PATTERN = '.*customers.*\.csv'
FILE_FORMAT = (FORMAT_NAME = RETAILPULSE_DB.BRONZE.CSV_FMT)
ON_ERROR = CONTINUE;
